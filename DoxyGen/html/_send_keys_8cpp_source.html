MIDI2LR  2.0.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

73  std::wstring utf8_to_utf16(const std::string& utf8) {
74  std::vector<unsigned long> unicode;
75  size_t i = 0;
76  while (i < utf8.size()) {
77  unsigned long uni;
78  size_t todo;
79  unsigned char ch = utf8[i++];
80  if (ch <= 0x7F) {
81  uni = ch;
82  todo = 0;
83  }
84  else if (ch <= 0xBF) {
85  throw std::domain_error("not a UTF-8 string");
86  }
87  else if (ch <= 0xDF) {
88  uni = ch & 0x1F;
89  todo = 1;
90  }
91  else if (ch <= 0xEF) {
92  uni = ch & 0x0F;
93  todo = 2;
94  }
95  else if (ch <= 0xF7) {
96  uni = ch & 0x07;
97  todo = 3;
98  }
99  else {
100  throw std::domain_error("not a UTF-8 string");
101  }
102  for (size_t j = 0; j < todo; ++j) {
103  if (i == utf8.size())
104  throw std::domain_error("not a UTF-8 string");
105  unsigned char ch = utf8[i++];
106  if (ch < 0x80 || ch > 0xBF)
107  throw std::domain_error("not a UTF-8 string");
108  uni <<= 6;
109  uni += ch & 0x3F;
110  }
111  if (uni >= 0xD800 && uni <= 0xDFFF)
112  throw std::domain_error("not a UTF-8 string");
113  if (uni > 0x10FFFF)
114  throw std::domain_error("not a UTF-8 string");
115  unicode.push_back(uni);
116  }
117  std::wstring utf16;
118  for (size_t i = 0; i < unicode.size(); ++i) {
119  unsigned long uni = unicode[i];
120  if (uni <= 0xFFFF) {
121  utf16 += (wchar_t)uni;
122  }
123  else {
124  uni -= 0x10000;
125  utf16 += (wchar_t)((uni >> 10) + 0xD800);
126  utf16 += (wchar_t)((uni & 0x3FF) + 0xDC00);
127  }
128  }
129  return utf16;
130  }

243 void SendKeys::SendKeyDownUp(const std::string& key, const bool alt_opt,
244  const bool control_cmd, const bool shift) const {
245  const auto mapped_key = SendKeys::key_map_.find(to_lower(key));
246  const auto in_keymap = mapped_key != SendKeys::key_map_.end();
247 
248 #ifdef _WIN32
249 
250  BYTE vk = 0;
251  BYTE vk_modifiers = 0;
252  if (in_keymap)
253  vk = mapped_key->second;
254  else {// Translate key code to keyboard-dependent scan code, may be UTF-8
255  const auto language_id = GetLanguage("Lightroom");
256  const auto vk_code_and_shift = VkKeyScanExW(MBtoWChar(key), language_id);
257  vk = LOBYTE(vk_code_and_shift);
258  vk_modifiers = HIBYTE(vk_code_and_shift);
259  }
260 
261  //construct virtual keystroke sequence
262  std::vector<size_t> strokes{vk}; // start with actual key, then mods
263  if (shift || (vk_modifiers & 0x1)) {
264  strokes.push_back(VK_SHIFT);
265  }
266  if ((vk_modifiers & 0x06) == 0x06) {
267  strokes.push_back(VK_RMENU); //AltGr
268  if (control_cmd)
269  strokes.push_back(VK_CONTROL);
270  if (alt_opt)
271  strokes.push_back(VK_MENU);
272  }
273  else {
274  if (control_cmd || (vk_modifiers & 0x2)) {
275  strokes.push_back(VK_CONTROL);
276  }
277  if (alt_opt || (vk_modifiers & 0x4)) {
278  strokes.push_back(VK_MENU);
279  }
280  }
281 
282  // construct input event.
283  INPUT ip;
284  constexpr auto size_ip = sizeof(ip);
285  ip.type = INPUT_KEYBOARD;
286  //ki: wVk, wScan, dwFlags, time, dwExtraInfo
287  ip.ki = {0,0,0,0,0};
288 
289  //send key down strokes
290  std::lock_guard<decltype(mutex_sending_)> lock(mutex_sending_);
291  for (auto it = strokes.crbegin(); it != strokes.crend(); ++it) {
292  ip.ki.wVk = static_cast<WORD>(*it);
293  SendInput(1, &ip, size_ip);
294  }
295  //send key up strokes
296  ip.ki.dwFlags = KEYEVENTF_KEYUP; // KEYEVENTF_KEYUP for key release
297  for (const auto it : strokes) {
298  ip.ki.wVk = static_cast<WORD>(it);
299  SendInput(1, &ip, size_ip);
300  }
301 #else
302  const CGEventSourceRef source =
303  CGEventSourceCreate(kCGEventSourceStateHIDSystemState);
304 
305  CGEventRef d;
306  CGEventRef u;
307 
308  CGEventRef cmdd = CGEventCreateKeyboardEvent(source, kVK_Command, true);
309  CGEventRef cmdu = CGEventCreateKeyboardEvent(source, kVK_Command, false);
310 
311  uint64_t flags = 0;
312 
313  if (in_keymap) {
314  const auto vk = mapped_key->second;
315  d = CGEventCreateKeyboardEvent(source, vk, true);
316  u = CGEventCreateKeyboardEvent(source, vk, false);
317  }
318  else {
319  const UniChar key_character(static_cast<UniChar>(utf8_to_utf16(key)[0]));
320  d = CGEventCreateKeyboardEvent(source, 0, true);
321  u = CGEventCreateKeyboardEvent(source, 0, false);
322  CGEventKeyboardSetUnicodeString(d, 1, &key_character);
323  CGEventKeyboardSetUnicodeString(u, 1, &key_character);
324  flags = CGEventGetFlags(d); //in case KeyCode has associated flag
325  }
326 
327  if (control_cmd) flags |= kCGEventFlagMaskCommand;
328  if (alt_opt) flags |= kCGEventFlagMaskAlternate;
329  if (shift) flags |= kCGEventFlagMaskShift;
330  if (flags) {
331  CGEventSetFlags(d, static_cast<CGEventFlags>(flags));
332  CGEventSetFlags(u, static_cast<CGEventFlags>(flags));
333  }
334 
335  constexpr CGEventTapLocation loc = kCGSessionEventTap;
336  if (flags & kCGEventFlagMaskCommand) {
337  std::lock_guard<decltype(mutex_sending_)> lock(mutex_sending_);
338  CGEventPost(loc, cmdd);
339  CGEventPost(loc, d);
340  CGEventPost(loc, u);
341  CGEventPost(loc, cmdu);
342  }
343  else {
344  std::lock_guard<decltype(mutex_sending_)> lock(mutex_sending_);
345  CGEventPost(loc, d);
346  CGEventPost(loc, u);
347  }
348 
349  CFRelease(d);
350  CFRelease(u);
351  CFRelease(cmdd);
352  CFRelease(cmdu);
353  CFRelease(source);
354 #endif
