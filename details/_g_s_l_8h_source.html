MIDI2LR  2.0.4.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

330  template< class T, class U >
331  gsl_api T narrow(U u) {
332  T t = narrow_cast<T>(u);
333 
334  if (static_cast<U>(t) != u) {
335  throw narrowing_error();
336  }
337 
338 #if gsl_HAVE_TYPE_TRAITS
339  if (!details::is_same_signedness<T, U>::value && ((t < T()) != (u < U())))
340 #else
341  // Don't assume T() works:
342  if ((t < 0) != (u < 0))
343 #endif
344  {
345  throw narrowing_error();
346  }
347  return t;
348  }

833  // ensure_sentinel()
834  //
835  // Provides a way to obtain a span from a contiguous sequence
836  // that ends with a (non-inclusive) sentinel value.
837  //
838  // Will fail-fast if sentinel cannot be found before max elements are examined.
839  //
840  namespace detail {
841  template<class T, class SizeType, const T Sentinel>
842  gsl_api static span<T> ensure_sentinel(T * seq, SizeType max = std::numeric_limits<SizeType>::max()) {
843  typedef T * pointer;
844 
845  pointer cur = seq;
846 
847  while (static_cast<SizeType>(cur - seq) < max && *cur != Sentinel)
848  ++cur;
849 
850  Expects(*cur == Sentinel);
851 
852  return span<T>(seq, cur - seq);
853  }
854  } // namespace detail

858  // Will fail fast if a null-terminator cannot be found before

863  gsl_api inline span<T> ensure_z(T * const & sz, size_t max = std::numeric_limits<size_t>::max()) {
864  return detail::ensure_sentinel<T, size_t, 0>(sz, max);
