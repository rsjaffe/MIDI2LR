MIDI2LR 6.2.0.1
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6+/CC Classic. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

42 enum struct MessageType : uint8_t {
43 kNoteOff = 0x8,
44 kNoteOn = 0x9,
45 kKeyPressure = 0xA, /* Individual key pressure */
46 kCc = 0xB,
47 kPgmChange = 0xC,
48 kChanPressure = 0xD, /* max key pressure */
49 kPw = 0xE, /* pitch wheel */
50 kSystem = 0xF
51 };
52
53 constexpr bool ValidMessageType(std::underlying_type_t<MessageType> value) noexcept
54 {
55 static_assert(std::is_unsigned_v<decltype(value)>, "Avoid sign extension");
56 const auto from {value >> 4U & 0xFU};
57 return std::cmp_greater_equal(from,
58 static_cast<std::underlying_type_t<MessageType>>(MessageType::kNoteOff));
59 }
60
61 constexpr MessageType ToMessageType(std::underlying_type_t<MessageType> value)
62 {
63 static_assert(std::is_unsigned_v<decltype(value)>, "Avoid sign extension");
64 const auto from {value >> 4U & 0xFU};
65 if (std::cmp_less(from,
66 static_cast<std::underlying_type_t<MessageType>>(MessageType::kNoteOff))) {
67 throw std::out_of_range("ToMessageType: MessageType range error, must be 0x8 to 0xF");
68 }
69 return static_cast<MessageType>(from);
70 }

72 inline const char* MessageTypeToName(MessageType from) noexcept
73 {
74 constexpr std::array translation_table {"Note Off", "Note On", "Key Pressure",

78 return translation_table[static_cast<size_t>(from)
79 - static_cast<size_t>(MessageType::kNoteOff)];

93template<typename Char> struct fmt::formatter<rsj::MessageType, Char> {
94 template<typename ParseContext> constexpr auto parse(ParseContext& ctx)
95 { /* parsing copied from fmt's chrono.h */
96 auto it {ctx.begin()};
97 if (!it) { return ctx.end(); }
98 if (it != ctx.end() && *it == ':') { std::advance(it, 1); }
99 auto end {std::find(it, ctx.end(), '}')};
100 tm_format_.reserve(detail::to_unsigned(end - it + 1));
101 tm_format_.append(it, end);
102 tm_format_.push_back('\0');
103 return end;
104 }
105
106 template<typename FormatContext> auto format(const rsj::MessageType& p, FormatContext& ctx) const
107 {
108 if (tm_format_[0] == 'n') {
109 return fmt::format_to(ctx.out(), "{}", rsj::MessageTypeToName(p));
110 }
111 return fmt::format_to(ctx.out(), "{}", rsj::MessageTypeToLabel(p));
112 }
113
114 private:
115 basic_memory_buffer<Char> tm_format_;
116}; // namespace fmt

144 /* channel is 0-based in MidiMessage, 1-based in MidiMessageId */
145 struct MidiMessageId {
146 int channel {1}; /* 1-based */
147 int control_number {0};
148 MessageType msg_id_type {MessageType::kNoteOn};
149
150 constexpr MidiMessageId() noexcept = default;
151
152 constexpr MidiMessageId(int ch, int dat, MessageType msgType) noexcept
153 : channel(ch), control_number(dat), msg_id_type(msgType)
154 {
155 }
156
157 explicit constexpr MidiMessageId(const MidiMessage& other) noexcept
158 : channel {other.channel + 1}, control_number {other.control_number},
159 msg_id_type {other.message_type_byte}
160 {
161 }
162#ifdef __cpp_lib_three_way_comparison
163 [[nodiscard]] constexpr std::strong_ordering operator<=>(
164 const MidiMessageId& other) const noexcept = default;
165#else
166 [[nodiscard]] constexpr bool operator==(const MidiMessageId& other) const noexcept
167 {
168 return msg_id_type == other.msg_id_type && channel == other.channel
169 && control_number == other.control_number;
170 }
171
172 [[nodiscard]] constexpr bool operator<(const MidiMessageId& other) const noexcept
173 {
174 if (channel < other.channel) { return true; }
175 if (channel == other.channel) {
176 if (control_number < other.control_number) { return true; }
177 if (control_number == other.control_number && msg_id_type < other.msg_id_type) {
178 return true;
179 }
180 }
181 return false;
182 }
183#endif
184 };
185} // namespace rsj

187/*It is allowed to add template specializations for any standard library class template to the
188 * namespace std only if the declaration depends on at least one program-defined type and the
189 * specialization satisfies all requirements for the original template, except where such
190 * specializations are prohibited. */
191template<> struct std::hash<rsj::MidiMessageId> {
192 size_t operator()(rsj::MidiMessageId k) const noexcept
193 {
194 /* channel is one byte, messagetype is one byte, controller (data) is two bytes */
195 constinit static std::hash<int_fast32_t> hasher;
196 return hasher(static_cast<int_fast32_t>(k.channel)
197 | static_cast<int_fast32_t>(k.msg_id_type) << 8
198 | static_cast<int_fast32_t>(k.control_number) << 16);
199 }
200};
